public class Main {
    public static int binarySearch(int[] a, int target) {
        int lo = 0, hi = a.length - 1;
        while (lo <= hi) {
            int mid = (lo + hi) >>> 1;
            if (a[mid] == target) return mid;

        }

```python
from typing import List
def quicksort(arr: List[int], lo: int=0, hi: int=None) -> None:
    if hi is None:
        hi = len(arr) - 1
    if lo < hi:
        p = partition(arr, lo, hi)
        quicksort(arr, lo, p - 1)
        quicksort(arr, p + 1, hi)

def partition(arr, lo, hi):
    pivot = arr[hi]
    i = lo
    for j in range(lo, hi):
        if arr[j] < pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    arr[i], arr[hi] = arr[hi], arr[i]
    return i

if __name__ == "__main__":
    a = [3,6,8,10,1,2,1]
    quicksort(a)
    print(a)

```python
class QueueTwoStacks:
    def __init__(self):
        self._in = []
        self._out = []

    def enqueue(self, x):
        self._in.append(x)

    def _move(self):
        while self._in:
            self._out.append(self._in.pop())

    def dequeue(self):
        if not self._out:
            self._move()
        if not self._out:
            raise IndexError('dequeue from empty queue')
        return self._out.pop()

```python
class QueueTwoStacks:
    def __init__(self):
        self._in = []
        self._out = []

    def enqueue(self, x):
        self._in.append(x)

    def _move(self):
        while self._in:
            self._out.append(self._in.pop())

    def dequeue(self):
        if not self._out:
            self._move()
        if not self._out:
            raise IndexError('dequeue from empty queue')
        return self._out.pop()

if __name__ == '__main__':
    q = QueueTwoStacks()
    q.enqueue(1); q.enqueue(2)
    print(q.dequeue(), q.dequeue())  # 1 2

from typing import List
def quicksort(arr: List[int], lo: int=0, hi: int=None) -> None:
    if hi is None:
        hi = len(arr) - 1
    if lo < hi:
        p = partition(arr, lo, hi)
        quicksort(arr, lo, p - 1)
        quicksort(arr, p + 1, hi)

def partition(arr, lo, hi):
    pivot = arr[hi]
    i = lo
    for j in range(lo, hi):
        if arr[j] < pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    arr[i], arr[hi] = arr[hi], arr[i]
    return i

    public static int binarySearch(int[] a, int target) {

```python
from typing import List
def quicksort(arr: List[int], lo: int=0, hi: int=None) -> None:
    if hi is None:
        hi = len(arr) - 1
    if lo < hi:
        p = partition(arr, lo, hi)
        quicksort(arr, lo, p - 1)
        quicksort(arr, p + 1, hi)

def partition(arr, lo, hi):
    pivot = arr[hi]
    i = lo
    for j in range(lo, hi):
        if arr[j] < pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    arr[i], arr[hi] = arr[hi], arr[i]
    return i

if __name__ == "__main__":
